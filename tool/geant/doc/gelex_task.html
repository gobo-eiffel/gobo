gelex

Description

Execute gelex.

Parameters

input
Description: input

Domain: Valid filename pointing to existing file

Default: -

output
Description: output

Domain: Valid filename

Default: -

array_size
Description: array_size

Domain: Integer

Default: 200

backup
Description: backup

Domain: -

Default: -

ecs
Description: ecs

Domain: -

Default: -

full
Description: full

Domain: -

Default: -

case_insensitive
Description: case_insensitive

Domain: -

Default: -

meta_ecs
Description: meta_ecs

Domain: -

Default: -

no_default
Description: no_default

Domain: -

Default: -

no_warn
Description: no_warn

Domain: -

Default: -

separate_actions
Description: separate_actions

Domain: -

Default: False

inspect_actions
Description: inspect_actions

Domain: -

Default: True

  <define name="gelex">
    <element name="gelex">
      <ref name="dir_if_unless"/>
      <attribute name="input"/>
      <optional>
        <attribute name="output"/>

      <optional>
        <attribute name="backup"/>
      </optional>
      <optional>
        <attribute name="array_size"/>

            <attribute name="meta_ecs"/>
          </optional>

      <optional>

      <optional>
        <attribute name="no_default"/>
      </optional>
      <optional>
        <attribute name="no_warn"/>
      </optional>
      <optional>
        <attribute name="separate_actions"/>
      </optional>
      <optional>
        <attribute name="inspect_actions"/>
      </optional>
    </element>
  </define>

  <gelex input="ascii2ps.l"/>
